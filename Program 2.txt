package program2:
public  class Car {


//instance members
 private String brd;  // brand
 private String mdl;// model
 private double prc; // price
 private double  mil;  // mileage
 private String  on; // ownerName

// static member
private static int   totalcars;       // totalCars
private static String showroomName = "town showroom";
// constructor 1
   public Car(String brd, String mdl, double prc, double mil) {
    this.brd="toyota"; // initialize brand
    this.mdl="suv"; // initialize model
    this.prc=2000000.d; // initialize price
    this.mil=27.d ; // initialize mileage
    this.on= "Not Assigned";
    totalcars++;
}
// constructor 2 (overloaded)
public Car(String brd, String mdl, double prc, double mil, String on) {
	this.brd="toyota";
    this.mdl="suv";
    this.prc=2000000.d;
    this.mil=27.d ;
    this.on= "owner";
    totalcars++;
}
// getter & setter for ownerName
public String getOwnerName(){
    return on;
}

public void setOwnerName(String owner) {
    this.on= "owner";
}

// instance method
public void displayDetails() {
    System.out.println("Brand: " + brd);
    System.out.println("Model: " +mdl);
    System.out.println("Price: " + prc);
    System.out.println("Mileage: " +mil);
    System.out.println("Owner: " + on);
}
public void updatePrice(double newPrice) {
   prc = newPrice;
}

// static methods
public static void showTotalCars() {
    System.out.println("Total Cars: " + totalcars);
}

public static void showShowroomName() {
    System.out.println("Showroom: " + showroomName);
}
}
